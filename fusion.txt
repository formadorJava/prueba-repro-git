	git commit -m "agregado fichero fusion.txt desde rama juan"

		git checkout main
		
	COGER LOS CAMBIOS DE LA RAMA juan Y GUARDARLOS EN LA RAMA ACTUALMENTE ACTIVA (main por la orden anterior).
		git merge juan
